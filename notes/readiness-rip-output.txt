let compute;

compute = function(questions) {
  let reviewEnabled, total;
  total = 0;
  for (const slug of ['a', 'b']) {
    if (slug === 'a') {
      (total++);
    } else {
      (total--);
    }
  }
  reviewEnabled = false;
  for (const q of questions) {
    if (q === 'x') {
      reviewEnabled = true;
      break;
    }
  }
  return {total, reviewEnabled};
};
